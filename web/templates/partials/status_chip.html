{{define "status_chip"}} {{- /* status_chip — rectangular status/metadata indicator. Shape: rectangle (--radius-sm). Pills are reserved for actions; see openspec/specs/ui-component-identity/spec.md (Shape language taxonomy). dict context: Kind string required — one of: billable, non-billable, running, draft, archived, warning Label string required — human-readable text Variant string required — "filled" or "outlined" Glyph string optional — leading glyph override; for state-conveying kinds (running, archived, draft, warning) a default glyph is rendered when omitted so state is never conveyed by color alone. */ -}} {{- $kind := .Kind -}} {{- $label := .Label -}} {{- $variant := .Variant -}} {{- $glyph := .Glyph -}} {{- if and (not $glyph) (eq $kind "running") -}}{{- $glyph = "●" -}}{{- end -}} {{- if and (not $glyph) (eq $kind "archived") -}}{{- $glyph = "⊘" -}}{{- end -}} {{- if and (not $glyph) (eq $kind "draft") -}}{{- $glyph = "○" -}}{{- end -}} {{/* U+26A0 U+FE0E (variation selector 15) forces text-style rendering on platforms that would otherwise substitute an emoji warning sign (which visually reads as ★ against dark backgrounds). */}} {{- if and (not $glyph) (eq $kind "warning") -}}{{- $glyph = "⚠︎" -}}{{- end -}} {{- if $glyph}}{{$glyph}}{{end -}} {{$label}} {{end}}
